fled_land_effect = {
	capital_scope = {
		remove_core = ROOT
		cede_province = previous_owner
		add_core = owner
		owner = {
			every_province = {
				limit = {
					same_continent = PREV
					is_core = ROOT
				}
				add_core = PREV
			}
			ROOT = {
				if = {
					limit = {
						religion_group = christian
					}
					PREV = {
						every_province = {
							limit = {
								same_continent = PREV
								is_core = ROOT
							}
							remove_core = ROOT
							cede_province = previous_owner
						}
					}
					create_union = PREV
					else = {
						PREV = {
							capital_scope = {
								cede_province = ROOT
								add_core = ROOT
							}
						}
					}
				}
			}
		}
	}
}

flee_effect = {
	change_tag = $TAG$
	every_subject_country = {
		limit = {
			is_colonial_nation = yes
			capital_scope = {
				colonial_region = $REGION$
			}
		}
		FROM = { inherit = PREV }
	}
	custom_tooltip = brazil_move_capital_tooltip

	hidden_effect = {
		fled_land_effect = yes
	}

	custom_tooltip = brazil_portugal_tooltip
	hidden_effect = {
		if = {
			limit = {
				$DEFAULT_CAPITAL$ = {
					owned_by = ROOT
				}
			}
			$DEFAULT_CAPITAL$ = {
				move_capital_effect = yes
			}
		}
		else = {
			random_owned_province = {
				limit = {
					colonial_region = $REGION$
				}
				move_capital_effect = yes
			}
		}          
	}
	remove_non_electors_emperors_from_empire_effect = yes
	add_stability = -2
	add_prestige = -25
	if = {
		limit = { has_custom_ideas = no }
		country_event = { id = ideagroups.1 }
	}
	set_country_flag = changed_from_colonial_nation
	set_country_flag = fled_to_brazil
}
